	<audio id="MyAudio" controls style="display: inline-block; width: 100%; aspect-ratio: 1/1;">
	<source src="lecturas.mp3" type="audio/mpeg">
	</audio>

    <script>
		function toggleAudio(button) {
		  var audio = document.getElementById("MyAudio");

		  if (audio.paused) {
			audio.play();
			button.classList.add("playing");
			button.innerHTML = '<b class="material-symbols-outlined">stop_circle</b>';
		  } else {
			audio.pause();
			button.classList.remove("playing");
			button.innerHTML = '<b class="material-symbols-outlined">play_circle</b>';
		  }
		}
    </script>
